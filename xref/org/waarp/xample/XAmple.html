39  /**
40   * A simple application demonstrates usage of the Swing-based XML editing component
41   * {@link com.fg.xmleditor.FXDoubleView}. The component generates document-specific GUI for each
42   * particular type of the XML document, specified by the XML Schema. A User-Manual document, which
43   * is included into XAmple XML Editor distribution package, provides detailed description of how to
44   * customize XML Schema documents with the editor-specific features, such as custom field editors,
45   * additional labeling, node-specific messages, etc.
46   * <p>
47   * </p>
48   * The application creares an instance of the {@link org.waarp.xample.custnodes.FFileDialog} and
49   * registers it with the {@link com.fg.xmleditor.FXDoubleView} under "FileDialog" id, so it can be
50   * used as a global field editor. (Actually there is no particular sense in doing it since the
51   * dialog is not initialized with any specific data. It is just an example).
52   * 
53   * @author Felix Golubov
54   * @author frederic bregier

107 	class TheFileFilter extends javax.swing.filechooser.FileFilter
108 	{
109 		String extension;
110 		String description;
111 
112 		public TheFileFilter(String extension, String description)
113 		{
114 			this.extension = extension;
115 			this.description = description;
116 		}
117 
118 		public boolean accept(File f)
119 		{
120 			if (f.isDirectory())
121 				return true;
122 			String ext = null;
123 			String name = f.getName();
124 			int i = name.lastIndexOf('.');
125 			if (i > 0 && i < name.length() - 1)
126 				ext = name.substring(i + 1).toLowerCase();
127 			else
128 				ext = "";
129 			return ext.equals(extension);
130 		}
131 
132 		public String getDescription()
133 		{
134 			return description;
135 		}
136 	}

277 	void saveRuntimeProperties()
278 	{
279 		Properties props = new Properties();
280 		LookAndFeel lf = UIManager.getLookAndFeel();
281 		if (lf != null)

286 		props.setProperty(BOUNDS_WIDTH, "" + r.width);
287 		props.setProperty(BOUNDS_HEIGHT, "" + r.height);

289 		try {
290 			File file = new File(FILE_RUNTIME);
291 			file.createNewFile();

293 			props.store(out, "XAmple-Waarp XML Editor runtime properties");
294 		} catch (Exception ex) {
295 			ex.printStackTrace();
296 		} finally {
297 			try {
298 				out.close();
299 			} catch (Exception ignore) {
300 			}
301 		}
302 		HistoryIO.save(history);

317 	public void itemStateChanged(ItemEvent e)
318 	{
319 		if (!confirmation())
320 			return;
321 		ItemSelectable source = e.getItemSelectable();
322 		String path = ((History) e.getItem()).path;
323 		File file = new File(path);
324 		if (source == btnOpenXSD)
325 			loadXMLSchema(file);
326 		else if (source == btnOpenXML)
327 			loadXMLDocument(file);
328 		fileDlg.setCurrentDirectory(file.getParentFile());
329 	}

331 	public void actionPerformed(ActionEvent e)
332 	{
333 		Object source = e.getSource();
334 		if (source == btnOpenXSD || source == mOpenXSD)
335 			openXSD();
336 		else if (source == btnOpenXML || source == mOpenXML)
337 			openXML();
338 		else if (source == btnNewXML || source == mNewXML)
339 			newXML();
340 		else if (source == btnSaveXML || source == mSaveXML)
341 			saveXML();
342 		else if (source == btnHorizSplit)
343 			dblView.setOrientation(JSplitPane.HORIZONTAL_SPLIT);
344 		else if (source == btnVertSplit)
345 			dblView.setOrientation(JSplitPane.VERTICAL_SPLIT);
346 		else if (source == btnSync)
347 			dblView.setSyncSelectNodes(btnSync.getModel().isSelected());
348 		else if (source instanceof JRadioButtonMenuItem)
349 			updateLookAndFeel(((JRadioButtonMenuItem) source).getText());
350 		else if (source == mAbout)
351 			showDlgAbout();
352 		else if (source == mExit)
353 			exit();
354 	}

356 	void updateLookAndFeel(String lfName)
357 	{
358 		LookAndFeel lf = UIManager.getLookAndFeel();
359 		if (lf != null && lf.getName().equals(lfName))
360 			return;
361 		UIManager.LookAndFeelInfo[] lfi = UIManager.getInstalledLookAndFeels();
362 		for (int i = 0; i < lfi.length; i++)
363 		{
364 			if (lfi[i].getName().equals(lfName))
365 			{
366 				try {
367 					// dblView.stopEditing();
368 					UIManager.setLookAndFeel(lfi[i].getClassName());
369 					SwingUtilities.updateComponentTreeUI(this);
370 					javax.swing.filechooser.FileFilter currFilter = fileDlg.getFileFilter();
371 					fileDlg.setAcceptAllFileFilterUsed(fileDlg.getAcceptAllFileFilter() == null);
372 					SwingUtilities.updateComponentTreeUI(fileDlg);
373 					fileDlg.setAcceptAllFileFilterUsed(true);
374 					fileDlg.setFileFilter(currFilter);
375 					SwingUtilities.updateComponentTreeUI(fFileDialog);
376 					return;
377 				} catch (Exception ignore) {
378 				}
379 			}
380 		}
381 	}

442 	void newXML()
443 	{
444 		if (!confirmation())
445 			return;
446 		try {
447 			if (model.getSchemaURL() == null)
448 				throw new FXModelException("No XML Configuration Schema loaded");
449 			NewDocumentDialog dlg = new NewDocumentDialog(this, model);
450 			dlg.pack();
451 			dlg.setLocationRelativeTo(dblView);
452 			dlg.setVisible(true);
453 			if (!dlg.isStatusOK())
454 				return;
455 			model.newDocument(model.getSchemaURL(), dlg.getRootNamespace(),
456 					dlg.getRootElementName());
457 			setXMLEnabled(true);
458 		} catch (Exception ex) {
459 			ex.printStackTrace();
460 			setXMLEnabled(false);
461 			dblView.showErrorMessage("Error: " + ex.getMessage());
462 			JOptionPane.showMessageDialog(this, "Can't open new XML Configuration Document",
463 					"Error", JOptionPane.ERROR_MESSAGE);
464 		}
465 		showXSDTitle();
466 	}

468 	boolean saveXML()
469 	{
470 		Document doc = model.getDocument();
471 		if (doc == null)
472 			return false;
473 		fileDlg.setFileFilter(xmlFilter);
474 		if (fileDlg.showSaveDialog(this) == JFileChooser.APPROVE_OPTION)
475 		{
476 			File file = fileDlg.getSelectedFile();
477 			if (file == null)
478 				return false;
479 			if (file.getName().indexOf(".") < 0)
480 				file = new File(file.getAbsolutePath() + ".xml");
481 			FileOutputStream out = null;
482 			Writer writer = null;
483 			try {
484 				file.createNewFile();
485 				out = new FileOutputStream(file);
486 				OutputFormat format = new OutputFormat(doc, "UTF-8", true);
487 				writer = new OutputStreamWriter(out, "UTF-8");
488 				XMLSerializer serial = new XMLSerializer(writer, format);
489 				serial.asDOMSerializer();
490 				serial.serialize(doc);
491 				model.setDocumentChanged(false);
492 				xmlFileName = file.getName();
493 				showXMLTitle();
494 				History childHistory = history.getFirstChild();
495 				if (childHistory != null)
496 				{
497 					childHistory.put(file.getAbsolutePath());
498 					btnOpenXML.setItems(childHistory.items);
499 				}
500 				return true;
501 			} catch (Exception ex) {
502 				dblView.showErrorMessage("Error: " + ex.getMessage());
503 				return false;
504 			} finally {
505 				try {
506 					out.close();
507 				} catch (Exception ignore) {
508 				}
509 				try {
510 					writer.close();
511 				} catch (Exception ignore) {
512 				}
513 			}
514 		}
515 		else
516 			return false;
517 	}

542 	boolean loadXMLSchema(File file)
543 	{
544 		try {
545 			model.newDocument(file.toURI().toURL());
546 			setSchemaEnabled(true);
547 			xsdFileName = file.getName();
548 			showXSDTitle();
549 			History childHistory = history.put(file.getAbsolutePath());
550 			btnOpenXSD.setItems(history.items);
551 			btnOpenXML.setItems(childHistory.items);
552 			dblView.showInfoMessage("XML Configuration Schema: " + file.getAbsolutePath());
553 			return true;
554 		} catch (Exception ex) {
555 			history.remove(file.getAbsolutePath());
556 			btnOpenXSD.setItems(history.items);
557 			btnOpenXML.setItems(null);
558 			dblView.showErrorMessage("Error: " + ex.toString());
559 			setSchemaEnabled(false);
560 			JOptionPane.showMessageDialog(this, "Can't load the XML Configuration Schema",
561 					"Error", JOptionPane.ERROR_MESSAGE);
562 			setTitle("XAmple");
563 			return false;
564 		}
565 	}

567 	boolean loadXMLDocument(File file)
568 	{
569 		History childHistory = history.getFirstChild();
570 		try {
571 			URL url = file.toURI().toURL();
572 			List lostElements =
573 					model.openDocument(model.getSchemaURL(), url);
574 			if (lostElements != null)
575 			{
576 				StringBuffer sb = new StringBuffer(
577 						"Error: The source XML Configuration document is invalid.\n" +
578 								"The following elements have not been loaded:");
579 				for (int i = 0; i < lostElements.size(); i++)
580 				{
581 					sb.append("\n");
582 					int k = sb.length();
583 					Node element = (Node) lostElements.get(i);
584 					sb.append(element.getNodeName());
585 					Node node = element.getParentNode();
586 					while (node != null && !(node instanceof Document))
587 					{
588 						sb.insert(k, node.getNodeName() + "/");
589 						node = node.getParentNode();
590 					}
591 				}
592 				dblView.showErrorMessage(sb.toString());
593 			}
594 			setXMLEnabled(true);
595 			xmlFileName = file.getName();
596 			showXMLTitle();
597 			childHistory.put(file.getAbsolutePath());
598 			btnOpenXML.setItems(childHistory.items);
599 			dblView.showInfoMessage("XML Configuration Document: " + file.getAbsolutePath());
600 			return true;
601 		} catch (Exception ex) {
602 			childHistory.remove(file.getAbsolutePath());
603 			btnOpenXML.setItems(childHistory.items);
604 			// dblView.clear();
605 			setXMLEnabled(false);
606 			dblView.showErrorMessage("Error: " + ex.getMessage());
607 			JOptionPane.showMessageDialog(this, "Can't load the XML Configuration Document",
608 					"Error", JOptionPane.ERROR_MESSAGE);
609 			showXSDTitle();
610 			return false;
611 		}
612 	}

622 	static void assignDefaultFont()
623 	{
624 		Properties props = new Properties();
625 		InputStream in = null;
626 		String fontName = null;
627 		int fontSize = 12;
628 		try {
629 			File file = new File(FILE_GUI);
630 			if (!file.exists())
631 				return;
632 			in = new FileInputStream(file);
