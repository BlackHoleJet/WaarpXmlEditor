47    class FileExtFilter extends javax.swing.filechooser.FileFilter
48    {
49      HashSet<String> extensions = new HashSet<String>();
50      String description = "";
51  
52      public FileExtFilter(String mask)
53      {
54        StringTokenizer st = new StringTokenizer(mask, "|");
55        while (st.hasMoreElements())
56        {
57          String token = st.nextToken().trim();
58          if (st.hasMoreElements()) {
59              if (token.length()==0) {
60                  // no extension
61              } else {
62                  if (token.startsWith("null")) {
63                      if (token.length()>4) {
64                          token = token.substring(5).trim();
65                      } else {
66                          token = "";
67                      }
68                  }
69                  if (token.length()>0) {
70                      extensions.add(token.toLowerCase());
71                  }
72              }
73          } else {
74              String type = token.substring(0, 1);
75              int val = Integer.parseInt(type);
76              if (val == 0) {
77                  mode = JFileChooser.FILES_ONLY;
78              } else if (val == 1) {
79                  mode = JFileChooser.DIRECTORIES_ONLY;
80              } else {
81                  mode = JFileChooser.FILES_AND_DIRECTORIES;
82              }
83              description = token.substring(1);
84          }
85        }
86        fileChooser.setFileSelectionMode(mode);
87      }
88  
89      public boolean accept(File f)
90      {
91        if (f.isDirectory())
92          return true;
93        if (f.isFile() && mode == JFileChooser.DIRECTORIES_ONLY)
94            return false;
95        if (extensions.size() == 0)
96            return true;
97        String ext = null;
98        String name = f.getName();
99        int i = name.lastIndexOf('.');
100       if (i > 0 &&  i < name.length() - 1)
101       {
102         ext = name.substring(i + 1).toLowerCase();
103         return extensions.contains(ext);
104       }
105       else
106         return false;
107     }

110     {
111       return description;
112     }

118     {

160   public void actionPerformed(ActionEvent e)

162     if (!(e.getSource() instanceof JButton))

164     JButton btn = (JButton)e.getSource();
165     if ("OK".equals(btn.getText()))
166     {
167       File file = fileChooser.getSelectedFile();
168       //data = (file == null || file.isDirectory()) ? "" : file.getAbsolutePath();
169       data = (file == null) ? "" : file.getAbsolutePath();
170       cancel();

173       cancel();

183   public void updateCellControl(boolean isEditor, boolean enabled,
184     boolean editable, Object data, Params params)
185   {
186     pack();
187     this.data = data;
188     if (data != null)
189     {
190       String path = data.toString().trim();
191       if (path.length() > 0)
192       {
193         try {
194           File file = new File(path);
195           fileChooser.setSelectedFile(file);
196         } catch (Exception ex) {
197           fileChooser.setSelectedFile(null);
198         }
199       }
200       else
201         fileChooser.setSelectedFile(null);
202     }
203     else
204       fileChooser.setSelectedFile(null);
205     List list = params.getList();
206     boolean sameMasks = true;
207     if (masks == null || list.size() != masks.size())
208       sameMasks = false;
209     else
210     {
211       for (int i = 0; i < list.size() && sameMasks; i++)
212       {
213         if (!list.get(i).equals(masks.get(i)))
214           sameMasks = false;
215       }
216     }
217     if (sameMasks)
218       return;
219     masks = list;
220     fileChooser.resetChoosableFileFilters();
221     for (int i = 0; i < masks.size(); i++)
222     {
223       fileChooser.addChoosableFileFilter(
224         new FileExtFilter((String)masks.get(i)));
225     }
226   }
